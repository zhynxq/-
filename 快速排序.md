#include<stdio.h>
#define KeyType int
#define OtherType int
#define MAX 100
typedef struct 
{
    KeyType key;
    OtherType other_data;
}RecordType;
int QKPass(RecordType r[],int low,int high)
{
    RecordType x;
    x=r[low];
    while(low<high)
    {
        while(low<high && r[high].key>=x.key)
            high--;
        if(low<high)
        {
            r[low]=r[high];
            low++;
        }
        while (low<high && r[low].key<x.key)
        {
            low++;
        }
        if(low<high)
        {
            r[high]=r[low];
            high--;
        }
    }
    r[low]=x;
    return low;
}
void QKSort(RecordType r[],int low,int high)
{
    int pos;
    if(low<high)
    {
        pos=QKPass(r,low,high);
        QKSort(r,low,pos-1);
        QKSort(r,pos+1,high);
    }
}
int main()
{
    RecordType r[MAX];
    int low,high;
    int i,len;
    printf("请输入需要排序的数字个数:\n");
    scanf("%d",&len);
    for(i=1;i<=len;i++)
    {
        printf("记录输入的第%i个元素",i);
        scanf("%d",&r[i].key);
    }
    printf("排序前的序列为:\n");
    for(i=1;i<=len;i++)
    {
        printf("%d ",r[i].key);
    }
    //快速排序
    QKSort(r,1,len);
    //排序后的结果
    printf("\n排序后的结果为:\n");
    for(i=1;i<=len;i++)
    {
        printf("%d ",r[i].key);
    }
    return 0;
}
